BiftVector

public struct BiftVector

A library for bit vectors (also called bit arrays) 

import BiftVector
let bv1 = ~BiftVector(size: 32)
let bv2 = BiftVector(uintVal: 557122642, size: 32)
let result = bv1 ^ bv2
let bv3 = BiftVector(hexString: "decafBAD")

if (result == bv3) {
